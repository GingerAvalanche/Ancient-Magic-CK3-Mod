	title:$TITLE$ = {
		if = {
			limit = {tier = tier_barony} 
			title_province = {
				add_province_modifier = ley_line_modifier
			}
		}
		if = {
			limit = {tier = tier_county} 
			every_in_de_jure_hierarchy = {
				limit = {
					tier = tier_barony
				}
				title_province = {
					add_province_modifier = ley_line_modifier
				}
			}
		}
	}
}
