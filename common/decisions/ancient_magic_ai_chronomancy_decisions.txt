haste_ai_spell_decision = {
	is_shown = {
		is_ai = yes
		is_magus = yes
		var:mana >= 100
		has_perk = mass_celerity_perk
	}
	
	ai_check_frequency = 1	
	
	effect = {
		if = {
			limit = {
				NOT = {
					has_character_modifier = magic_haste_modifier
				}
			}

			change_mana = { VALUE = -100 }

			change_mana_gen = { VALUE = -15 }

			while = {
				count = mage_level
				add_character_modifier = {
					modifier = magic_haste_modifier
					mode = add
				}
			}
		}
		else_if = {
			limit = {
				OR = {
					has_character_modifier = magic_haste_modifier
				}
			}
			change_mana_gen = { VALUE = 15 }

			while = {
				count = 100
				remove_character_modifier = magic_haste_modifier
			}
		}
	}
	
	ai_potential = { always = yes }
	
	ai_will_do = {
		base = 25
		
        ai_value_modifier = {
            ai_rationality = 0.2
        }
			
		modifier = {
			factor = 0
			var:mana_gen < 15
		}
		
		modifier = {
			factor = 0
			has_character_modifier = magic_haste_modifier
		}
		
		modifier = {
			add = 50
			has_character_modifier = magic_haste_modifier
			var:mana_gen <= 0
		}
	}
}
