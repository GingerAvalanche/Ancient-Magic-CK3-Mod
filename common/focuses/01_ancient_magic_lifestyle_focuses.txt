magic_arcana_focus = {
	lifestyle = arcane_lifestyle
	
	focus_id = 101
	
	desc = {
		desc = magic_arcana_focus_desc
		desc = line_break
	}
	
	modifier = {
		monthly_arcane_lifestyle_xp_gain_mult = 0.2
	}
	
	on_change_to = {
		trigger_event = { on_action = on_reset_mana_system }
	}
	
	on_change_from = {
		trigger_event = { on_action = on_reset_mana_system }
	}

	auto_selection_weight = {
		value = 11
		if = {
			limit = {
				has_trait = magic_potential	
			}
			add = 1989
		}
		if = {
			limit = {
				has_trait = ambitious
			}
			multiply = 2;
		}
		if = {
			limit = {
				has_trait = diligent
			}
			multiply = 2;
		}
		if = {
			limit = {
				has_trait = content
			}
			add = -300
		}
	}
}
