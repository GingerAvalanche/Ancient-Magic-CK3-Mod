is_duchy_settings_toolbar_open = {
	scope = character
	is_valid = {
		has_variable = toolbar_duchy_settings_view
	}
}

toggle_duchy_settings_toolbar = {
	scope = character
	effect = {
		if = {
			limit = {
				has_variable = toolbar_duchy_settings_view
			}
			effect_close_duchy_settings_toolbar = yes
		}
		else = {
			effect_open_duchy_settings_toolbar = yes
		}
	}
}
